99 void Instrument<T, Targs...>::msg_handler() {
100 
101  if (m_msg == midi::note_on && m_msgdata[1] == 0)
102  m_msg = midi::note_off;
103  if (m_msg == midi::note_on) {
104  for (auto &note : m_voices) {
105  if (!note.is_on()) {
106  note.note_on(m_chn, m_msgdata[0], m_msgdata[1], m_stamp);
107  return;
108  }
109  }
110  auto ostamp = m_voices[0].time_stamp();
111  auto oldest = &m_voices[0];
112  for (auto &note : m_voices) {
113  if (note.time_stamp() < ostamp) {
114  oldest = &note;
115  ostamp = note.time_stamp();
116  }
117  }
118  oldest->note_off();
119  oldest->note_on(m_chn, m_msgdata[0], m_msgdata[1], m_stamp);
120  } else if (m_msg == midi::note_off) {
121  for (auto &note : m_voices) {
122  note.note_off(m_chn, m_msgdata[0], m_msgdata[1]);
123  }
124  } else {
125  for (auto &note : m_voices) {
