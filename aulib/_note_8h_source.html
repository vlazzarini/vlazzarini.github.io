181 template <typename T, typename... Targs>
182 void Instrument<T, Targs...>::msg_handler() {
183 
184  if (m_msg == midi::note_on && m_msgdata[1] == 0)
185  m_msg = midi::note_off;
186  if (m_msg == midi::note_on) {
187  for (auto &note : m_voices) {
188  if (!note.is_on()) {
189  note.note_on(m_chn, m_msgdata[0], m_msgdata[1], m_stamp);
190  return;
191  }
192  }
193  auto ostamp = m_voices[0].time_stamp();
194  auto oldest = &m_voices[0];
195  for (auto &note : m_voices) {
196  if (note.time_stamp() < ostamp) {
197  oldest = &note;
198  ostamp = note.time_stamp();
199  }
200  }
201  oldest->note_off();
202  oldest->note_on(m_chn, m_msgdata[0], m_msgdata[1], m_stamp);
203  } else if (m_msg == midi::note_off) {
204  for (auto &note : m_voices) {
205  note.note_off(m_chn, m_msgdata[0], m_msgdata[1]);
206  }
207  } else {
208  for (auto &note : m_voices) {
209  note.ctrl_msg(m_chn, m_msg, m_msgdata, m_stamp);
210  }
211  }
212 }
