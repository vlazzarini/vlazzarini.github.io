81  virtual void parse_stream(std::istream &input, uint64_t offset = 0) {
82  std::string cmd;
83  double time;
84  Event ev;
85  while (input >> cmd) {
86  for (auto &c : m_cmds) {
87  if (c.cmd == cmd) {

154  template <typename... Targs> const Score &process(Targs &... args) {
155  if (!m_done) {
156  auto it = m_events.begin();
157  for (auto &ev : m_events) {
158  if (m_is_sorted && ev.time >= m_time)
159  break;
160  if (m_time > ev.time) {
161  if (ev.msg == 0) {
162  m_done = true;
163  break;
164  }
165  dispatch(ev, args...);
166  it = m_events.erase(it);
167  } else std::advance(it, 1);
168  }
169  }
170  set(0.);
171  dsp(args...);
172  m_time += vframes();
173  return *this;
174  }
